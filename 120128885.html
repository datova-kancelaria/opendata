Metodika pre otvorené údaje (opendata.gov.sk) : Q: Datasety a ich poskytovatelia

Created by miroslav.liska, last modified on feb 13, 2024 
PREFIX dcat: <http://www.w3.org/ns/dcat#>
PREFIX dct: <http://purl.org/dc/terms/>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>
 
SELECT DISTINCT ?dataset ?názov ?poskytovateľ WHERE {
  GRAPH ?g {
    ?dataset a dcat:Dataset;
      dct:title ?názov;
      dct:publisher/foaf:name ?poskytovateľ.
      FILTER(langMatches(LANG(?názov), "sk"))
      FILTER(langMatches(LANG(?poskytovateľ), "sk"))
  }
}
ORDER BY ?názov
