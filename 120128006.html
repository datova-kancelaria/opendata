Metodika pre otvorené údaje (opendata.gov.sk) : Zoznam URI poskytovateľov a názov poskytovateľa

Created by Lukáš Jankovič on apr 19, 2024

PREFIX dcat: <http://www.w3.org/ns/dcat#>
PREFIX dct: <http://purl.org/dc/terms/>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>

SELECT DISTINCT ?poskytovateľ ?poskytovateľ1 WHERE {
  GRAPH ?g {
    ?dataset a dcat:Dataset;
      dct:publisher ?poskytovateľ ;
      dct:publisher/foaf:name ?poskytovateľ1 .
   }
}
