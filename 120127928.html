Metodika pre otvorené údaje (opendata.gov.sk) : Meranie dátovej kvality pomocou SPARQL

Created by Lukáš Jankovič, last modified on aug 01, 2025

Dotazy kontrolujú vyplnenie povinných metaúdajov (či nechýbajú povinné metaúdaje), nekontrolujú však či majú validnú hodnotu (napr. podľa číselníka), či je vyplnenie správne.

NeHVD datasety

Povinné metadáta datasetu a distribúcie - súbor na stiahnutie

PREFIX dcat: <http://www.w3.org/ns/dcat#>
PREFIX dct: <http://purl.org/dc/terms/>

SELECT ?dataset ?názov ?téma ?kľúčovésl
WHERE {
    ?dataset a dcat:Dataset.
    ?dataset dct:title ?názov.
    ?dataset dct:publisher <https://data.gov.sk/id/legal-subject/50349287>.
    OPTIONAL {?dataset dcat:theme ?téma. }
  	OPTIONAL {?dataset dcat:keyword ?kľúčovésl. }
}

PREFIX dcat: <http://www.w3.org/ns/dcat#>
PREFIX dct: <http://purl.org/dc/terms/>
PREFIX leg: <https://data.gov.sk/def/ontology/legislation/> 

SELECT ?dataset ?názov ?periodicita ?uzpokrytie ?dist ?url_stiahnutie ?url_prístupové ?formát ?typ_média ?podmienky_použitia
WHERE {
    ?dataset a dcat:Dataset.
    ?dataset dct:title ?názov.
    ?dataset dct:publisher <https://data.gov.sk/id/legal-subject/50349287>.
    OPTIONAL {?dataset dct:accrualPeriodicity ?periodicita. }
    OPTIONAL {?dataset dct:spatial ?uzpokrytie. }
    ?dataset dcat:distribution ?dist.
    OPTIONAL {?dist dcat:downloadURL ?url_stiahnutie. }
    OPTIONAL {?dist dcat:accessURL ?url_prístupové. }
    OPTIONAL {?dist dct:format ?formát. }
    OPTIONAL {?dist dcat:mediaType ?typ_média. }
    OPTIONAL {?dist leg:termsOfUse ?podmienky_použitia. }
    FILTER NOT EXISTS { ?dist dcat:accessService ?service. }
}

PREFIX dcatap: <http://data.europa.eu/r5r/>
PREFIX dcat: <http://www.w3.org/ns/dcat#>
PREFIX dct: <http://purl.org/dc/terms/>
PREFIX leg: <https://data.gov.sk/def/ontology/legislation/> 

SELECT ?dataset ?názov ?periodicita ?uzpokrytie ?právny_predpis ?kat_HVD ?dist ?url_stiahnutie ?url_prístupové ?formát ?typ_média ?podmienky_použitia ?právny_predpis_dist
WHERE {
    ?dataset a dcat:Dataset.
    ?dataset dct:title ?názov.
    ?dataset dct:type <http://publications.europa.eu/resource/authority/dataset-type/HVD>.
    ?dataset dct:publisher <https://data.gov.sk/id/legal-subject/50349287>.
    OPTIONAL {?dataset dct:accrualPeriodicity ?periodicita. }
    OPTIONAL {?dataset dct:spatial ?uzpokrytie. }
    OPTIONAL {?dataset dcatap:applicableLegislation ?právny_predpis. }
    OPTIONAL {?dataset dcatap:hvdCategory ?kat_HVD. }
    ?dataset dcat:distribution ?dist.
    OPTIONAL {?dist dcat:downloadURL ?url_stiahnutie. }
    OPTIONAL {?dist dcat:accessURL ?url_prístupové. }
    OPTIONAL {?dist dct:format ?formát. }
    OPTIONAL {?dist dcat:mediaType ?typ_média. }
    OPTIONAL {?dist leg:termsOfUse ?podmienky_použitia. }
    OPTIONAL {?dist dcatap:applicableLegislation ?právny_predpis_dist. }
    FILTER NOT EXISTS { ?dist dcat:accessService ?service. }
}
